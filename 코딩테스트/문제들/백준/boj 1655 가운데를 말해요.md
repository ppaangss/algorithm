# 회고

시간복잡도가 핵심인 문제이다. 

이 문제는 단순히 입력할 때 마다 중간값을 고르면 되는 문제다. 그래서 단순히 들어올 때 마다 정렬을 하면 되지 않을까? 라는 생각을 했다. 

하지만 시간복잡도를 고려한다면 말이 안되는 상황이었다. 일단 시간은 0.1초로 약 1억번 내에 연산을 수행해야 한다. 하지만 정렬을 한다면 총 $O(N^{2}logN)$ 의 시간복잡도로 최악의 경우 N이 100,000 이면 시간초과가 발생하게 된다.

그래서 문득 든 생각이 sort 라이브러리는 퀵 정렬이고, 퀵 정렬은 정렬되어있는 상태에서 정렬을 시도하면 시간복잡도가 오히려 $O(N^2)$ 에 가까워진다는 것을 알았다.

그래서 삽입정렬을 생각했다. 삽입정렬은 거의 정렬이 되어있는 상태일 때 시간복잡도가 $O(N)$ 의 성능을 갖는 것을 알고 있었다. 그래서 시도했는데 역시나 안됐다. ( 그리고 이제 와서 안 것이지만 $O(N)$의 성능으로도 해결 못한다 $O(logN)$ 이 어야 한다. )

삽입정렬이 시간복잡도로 이득을 볼 상황은 앞에 있는 데이터들이 정렬이 되어있는 상태를 의미하는 것이 아니라 이미 정렬이 되어있을 떄 유리한 것이다. 그니까, 새로운 데이터가 들어왔을 때 이동이 없어야 유리한 것이다. 하지만 지금 경우는 다르다. 새로운 데이터가 계속 들어올 경우 이동이 계속해서 생길 수 밖에 없는 문제이다.

결국 이 문제는 우선순위 큐를 사용하는 것이다. 근데 우선순위 큐를 어떻게 사용해야 할지 막막했다. 우선순위를 중간값으로 정해줄 수 가 있나?

정답은 우선순위 큐를 2개 쓰는 것이다. 최대힙과 최소힙 2개 , 그리고 두 힙의 개수를 계속해서 맞춰주면 각 힙의 값은 중간값이 될 수 밖에 없다.

이것도 규칙성과 예외로 인해 코드가 많이 복잡해졌다. 깔끔하게 짜는 연습을 좀 해야할 것 같다.

# 풀이

### 정렬 (실패)

```cpp
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>


using namespace std;


int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;
	vector<int> vt;
	vector<int> res;
	for (int i = 1; i <= N; i++) {
		//cout << "i-" << i << "\n";
		int tmp; cin >> tmp;
		vt.push_back(tmp);

		sort(vt.begin(), vt.end());
		
		int n1 = vt[(i-1)/2];
		//cout << "n1-" << n1 << "\n";
		if (i % 2 == 0) {
			int n2 = vt[(i-1) / 2 + 1];
			//cout << "n2-" << n2 << "\n";
			n1 = min(n1, n2);
		}
		//cout << "\n";

		res.push_back(n1);

	}

	for (const auto& x : res) {
		cout << x << "\n";
	}
	
	return 0;
}
```

```cpp
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>


using namespace std;


int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;
	vector<int> vt;
	vector<int> res;
	for (int i = 1; i <= N; i++) {
		//cout << "i-" << i << "\n";
		int tmp; cin >> tmp;

		bool flag = true;
		int size1 = vt.size();
		for (int j = 0; j < size1; j++) {
			//cout << vt[j] << "\n";
			if (vt[j] > tmp) {
				flag = false;
				vt.insert(vt.begin() + j, tmp);
				break;
			}
		}

		if (flag) {
			vt.push_back(tmp);
		}

		int n1 = vt[(i-1)/2];
		//cout << "n1-" << n1 << "\n";
		if (i % 2 == 0) {
			int n2 = vt[(i-1) / 2 + 1];
			//cout << "n2-" << n2 << "\n";
			n1 = min(n1, n2);
		}
		//cout << "\n";

		res.push_back(n1);

	}

	for (const auto& x : res) {
		cout << x << "\n";

	return 0;
```

### 우선순위 큐 (성공)

```cpp
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <queue>

using namespace std;

// res 값 반환
int FUNC(int num, priority_queue<int>& pq_big, priority_queue<int, vector<int>, greater<int>>& pq_sml){

	pq_big.push(num);

	// big에 들어가야하냐, sml에 들어가야 하냐

	if (pq_sml.empty()) {
		pq_sml.push(pq_big.top());
		pq_big.pop();
	}
	else if (pq_big.top() > pq_sml.top()) {
		pq_sml.push(pq_big.top());
		pq_big.pop();
	}
	// 개수 비교
	
	//while
	if (pq_big.size() - pq_sml.size() == 2) {
		pq_sml.push(pq_big.top());
		pq_big.pop();
	}
	else if (pq_big.size() - pq_sml.size() == -2) {
		pq_big.push(pq_sml.top());
		pq_sml.pop();
	}

	// 무조건 2개 중 하나 , 짝수냐 홀수냐
	if (pq_big.size() - pq_sml.size() == 1) {
		//cout << "big-" << pq_big.top() << "\n";
		//cout << "sml-" << pq_sml.top() << "\n";

		return pq_big.top();
	}
	else if (pq_big.size() - pq_sml.size() == -1) {
		//cout << "big-" << pq_big.top() << "\n";
		//cout << "sml-" << pq_sml.top() << "\n";

		return pq_sml.top();
	}
	else if (pq_big.size() - pq_sml.size() == 0) {
		//cout << "big-" << pq_big.top() << "\n";
		//cout << "sml-" << pq_sml.top() << "\n";
		return min(pq_big.top(),pq_sml.top());
	}
}

int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;
	//vector<int> vt;

	priority_queue<int> pq_big;
	priority_queue<int, vector<int>, greater<int>> pq_sml;
	vector<int> res; // 결과

	int tmp1; cin >> tmp1;
	pq_big.push(tmp1);
	res.push_back(tmp1);

	for (int i = 1; i < N; i++) {
		//cout << "i-" << i << "\n";
		int tmp2; cin >> tmp2;

		int res1 = FUNC(tmp2, pq_big, pq_sml);

		//cout << "res1-" << res1 << "\n";
		res.push_back(res1);
	}

	for (const auto& x : res) {
		cout << x << "\n";
	}
	
	return 0;
}
```
