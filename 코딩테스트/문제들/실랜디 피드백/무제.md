#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <queue>
using namespace std;

int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;

	if (N == 1) {
		cout << 1 << "\n";
		return 0;
	}

	int x = 1;
	int start = 2;
	int size1 = 6; // 겉에 개수
	while (1) {
		//cout << "범위" << "\n";
		//cout << start <<  "-" << start + size1 << "\n";
		if (start <= N && start + size1 > N) {
			cout << x+1 << "\n";
			break;
		}
		x++;

		start += size1;
		size1 = size1 + (x ) * 2;

	}



	return 0;
}
