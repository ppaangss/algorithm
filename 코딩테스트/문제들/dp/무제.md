#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    // 입출력 속도 향상
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N; cin >> N;


    vector<vector<int>> dp(N + 1, vector<int>(3, 0));

    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            cin >> dp[i][j];
        }
    }

    for (int i = 1; i <= N; i++) {
        dp[i][0] += min(dp[i - 1][1], dp[i - 1][2]);
        dp[i][1] += min(dp[i - 1][0], dp[i - 1][2]);
        dp[i][2] += min(dp[i - 1][0], dp[i - 1][1]);
    }

    cout << min({ dp[N][0],dp[N][1],dp[N][2] }) << "\n";

    return 0;
}
