# 회고

일반적인 BFS와는 다르게 이 경우는 한 번 방문한 곳을 다시 막아 서는 안된다.

왜냐하면 해당 도착하는 경우까지의 방법의 수를 구해야 하므로 재 방문할 수 있느 경우가 있기 때문이다. 

두 가지 형태로 풀어볼 수 있겠다. 

첫번째 방식은 똑같이 하되, visited 방문을 pop 할 때 처리하는 것이다. 이렇게 하면 직관적으로 여러 방문을 확인할 수 있다. 

https://hagisilecoding.tistory.com/104

두번째 방식은 각 노드마다의 최소 도착 시간과 방문 횟수를 배열에 저장하는 것이다. 

https://eunsolsblog.tistory.com/entry/C-%EB%B0%B1%EC%A4%80-12851%EB%B2%88-%EC%88%A8%EB%B0%94%EA%BC%AD%EC%A7%88-2

어느 방식을 하던 상관은 없으나 결국 풀면 장떙이다. 난 못풀었다.

```cpp
#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

//
#include <queue>

using namespace std;

bool visited[100001];

int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;
	int K; cin >> K;

	queue<pair<int, int>> qu;

	int time1 = 0;
	int cnt1 = 0;
	bool flag = false;

	qu.push({ N,1 });
	visited[N] = true;


	while (!qu.empty()) {

		int cx = qu.front().first;
		int cNum = qu.front().second;
		qu.pop();

		visited[cx] = true;

		if (flag) {
			if (cx == K) {
				if (cNum == time1) {
					cnt1++;
				}
			}
		}
		else {
			if(cx == K) {
				flag = true;
				cnt1++;
				time1 = cNum;
			}
		}


		int dx[3] = { cx - 1,cx + 1,cx * 2 };

		int nNum = cNum + 1;

		for (int i = 0; i < 3; i++) {
			int nx = dx[i];
			if (nx >= 0 && nx <= 100000) {
				if (visited[nx] == 0) { // 한 번도 방문을 안했거나, 
					qu.push({ nx,nNum });
				}
			}
		}
	}

	cout << time1-1 << "\n";
	cout << cnt1 << "\n";

	return 0;
}
```

```cpp
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;

int dist[100002];
int cnt[100002];
queue<int> q;

int main(void)
{
	ios::sync_with_stdio(0);
	cin.tie(0);

	int n, k;
	cin >> n >> k;

	if (n == k)
	{
		cout << 0 <<"\n"<<1;
		return 0;
	}

	fill(dist, dist + 100002, -1);

	dist[n] = 0;
	cnt[n] = 1;
	q.push(n);

	while (!q.empty())
	{
		int cur = q.front(); q.pop();
		for (int nx : {cur + 1, cur - 1, cur * 2})
		{
			if (nx < 0 || nx>100000) continue;
			if (dist[nx] != -1)	//방문 했던 곳이면
			{
				if (dist[cur] + 1 == dist[nx])//최단 거리가 같은 경우
					cnt[nx] += cnt[cur];
			}
			else
				//방문을 처음 했다면
			{
				dist[nx] = dist[cur] + 1;	//최단 거리 갱신
				cnt[nx] = cnt[cur];			//처음 간 경우
				q.push(nx);
			}
		}
	}

	cout << dist[k] << "\n" << cnt[k];

	return 0;
}
```
