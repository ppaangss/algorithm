## 포인터 없이 일반 트리

```c++
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>

// 컨테이너
#include <queue>

#define INF 87654321

using namespace std;

struct Node {
    int data;
    std::vector<int> children; // 자식 노드들의 인덱스를 가변적으로 저장
};

vector<Node> tree;

// 새 노드를 생성하고 인덱스를 반환
int create_node(int val) {
    Node newNode;
    newNode.data = val;
    // children 벡터는 비어있는 상태로 시작
    tree.push_back(newNode);
    return tree.size() - 1;
}

// 부모 노드에 자식 노드를 연결하는 함수
void add_child(int parent_idx, int child_idx) {
    if (parent_idx == -1) return;
    tree[parent_idx].children.push_back(child_idx);
}

// 일반 트리 순회 (DFS)
void traverse(int curr_idx, int depth) {
    if (curr_idx == -1) return;

    // 가독성을 위해 들여쓰기 출력
    for (int i = 0; i < depth; i++) cout << "  ";
    cout << "- " << tree[curr_idx].data << "\n";

    // 모든 자식을 순회
    for (int child_idx : tree[curr_idx].children) {
        traverse(child_idx, depth + 1);
    }
}

int main() {
    // 1. 노드 생성 (인덱스 확보)
    int n50 = create_node(50);
    int n30 = create_node(30);
    int n98 = create_node(98);
    int n24 = create_node(24);
    int n45 = create_node(45);
    int n52 = create_node(52);
    int n5 = create_node(5);
    int n28 = create_node(28);
    int n60 = create_node(60);

    // 2. 부모-자식 연결 (이미지 기준)
    add_child(n50, n30); add_child(n50, n98); // 50의 자식
    add_child(n30, n24); add_child(n30, n45); // 30의 자식
    add_child(n24, n5);  add_child(n24, n28); // 24의 자식
    add_child(n98, n52);                      // 98의 자식
    add_child(n52, n60);                      // 52의 자식

    // 3. 루트(50번)부터 출력
    cout << "--- 일반 트리 구조 출력 ---\n";
    traverse(n50, 0);

    return 0;
}
```

## 포인터 없는 이진 탐색 트리

```c++
#include <iostream>

struct Node {
    int data;
    int left;  // 왼쪽 자식의 '인덱스' (포인터 대신)
    int right; // 오른쪽 자식의 '인덱스'
};

Node tree[100001]; // 최대 노드 수만큼 미리 할당
int node_count = 0; // 현재까지 생성된 노드 개수 (주소 역할)

// 새 노드를 생성하고 인덱스를 반환하는 함수
int create_node(int val) {
    node_count++;
    tree[node_count].data = val;
    tree[node_count].left = -1;  // -1은 NULL을 의미
    tree[node_count].right = -1;
    return node_count;
}

// BST 삽입 함수 (재귀 버전)
int insert(int root_idx, int val) {
    // 1. 빈 자리에 도달하면 새 노드 생성
    if (root_idx == -1) {
        return create_node(val);
    }

    // 2. 값 비교 후 인덱스 연결
    if (val < tree[root_idx].data) {
        tree[root_idx].left = insert(tree[root_idx].left, val);
    } else {
        tree[root_idx].right = insert(tree[root_idx].right, val);
    }

    return root_idx; // 현재 루트 인덱스 반환
}

int main() {
    int root = -1; // 처음엔 루트가 없음 (NULL 대용)

    root = insert(root, 50);
    insert(root, 30);
    insert(root, 70);

    // 이제 tree[root]를 시작으로 인덱스를 따라가며 탐색 가능
    return 0;
}
```
