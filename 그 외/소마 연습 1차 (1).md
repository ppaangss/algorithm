#include <iostream>
#include <algorithm>
#include <vector>
#include <unordered_set>

using namespace std;

int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int N; cin >> N;

	vector<vector<long long>>dp(N+1, vector<long long>(3, 0));

	// 0은 오른쪽에 사자가 있다.
	// 1은 사자가 없다.
	// 2는 왼쪽에 사자가 있다.
	dp[1][0] = 1;
	dp[1][1] = 1;
	dp[1][2] = 1;

	for (int i = 2; i <= N; i++) {
		dp[i][0] = dp[i - 1][1] + dp[i - 1][2];
		dp[i][1] = dp[i - 1][0] + dp[i - 1][1] + dp[i - 1][2];
		dp[i][2] = dp[i - 1][0] + dp[i - 1][1];
	}

	long long sum = 0;
	for (int i = 0; i < 3; i++) {
		int tmp = dp[N][i] % 9901;
		sum += tmp;
		sum %= 9901;
	}

	cout << sum << "\n";

	return 0;
}
